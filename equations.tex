    \item Common Pose Operations:
    \begin{align*}
        P^A_B \cdot P^B_C &= P^A_C \\
        \left(P^A_B\right)^{-1} &= P^B_A \\
        T^A_B \cdot v_A &= v_B \\
    \end{align*}

    \item Position Operations:
    \begin{align*}
        \begin{bmatrix}

            x \\ y

        p^A_B + p^B_C &= p^A_C \\
        -p^A_B &= p^B_A \\
        p^A_B + p_A &= p_B \\
    \end{align*}

    \item commutative:
    \begin{align*}
        R_1 \cdot R_2 &= R_2 \cdot R_1 \\
        R_1 \cdot R_2 &\neq R_2 \cdot R_1 \\
    \end{align*}

    \item Rotation Matrix:

            \end{bmatrix} & \begin{bmatrix}
